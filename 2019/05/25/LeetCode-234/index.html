不行的代码

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    bool isPalindrome(ListNode* head) {
        if(head == nullptr || head->next == nullptr)
            return true;
        int res = 0;
        int flag = 0;
        ListNode * list = head;
        int tmm = 0;
        while(list != nullptr){
            if(flag == 0)
                res = res*10+ list->val;
            else{
                tmm = res%10;
                res /= 10;
                if(tmm != list->val){
                    return false;
                }
                if(res == 0){
                    return true;
                }
            }
            if(flag == 0 && list->next == nullptr)
                return false;
            if(flag == 0&&flag == 0){
                flag =1 ;
            }
            list = list->next;
        }
        return false;
    }
};
会卡在 [-129,129] 处，因为该算法都是按照链表元素为个数算的。

错误实例2：

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    bool isPalindrome(ListNode* head) {
        if(head == nullptr || head->next == nullptr)
            return true;
        stack<int> stk;
        int flag = 0;
        ListNode * list = head;
        while(list != nullptr){
            //flag是0，表示还没遇到那个开始回文的地方。
            if(flag == 0)
                stk.push(list->val);
            else{
                if(stk.empty())
                    return false;
                int tmp = stk.top();
                stk.pop();
                if(tmp != list->val)
                    return false;
            }
            if(flag == 0 && list->next == nullptr)
                return false;
            if(flag == 0&& list->val == list->next->val){
                flag =1 ;
            }
            list = list->next;
        }
        if(stk.empty())
            return true;
        return false;
    }
};
会卡在[1,1,2,2,1,1]

ok:

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    bool isPalindrome(ListNode* head) {
        ListNode * node = head;
        stack<int> stk1;
        while(node != nullptr){
            stk1.push(node->val);
            node = node->next;
        }
        node = head;
        while(node != nullptr){
            if(stk1.top() != node->val)
                return false;
            stk1.pop();
            node = node->next;
        }
        if(stk1.empty())
            return true;
        return false;
    }
};
