排序双指针法:

先讲一讲C++ sort函数的用法

对vector排序

sort(nums1.begin(),nums1.end());

对数组排序也是一样的用法，第一个参数为起始地址，第二个参数为最后一个元素的后一位的地址。

因为sort实际上是不去第二个参数所表明的地址的值的，就和vector.end()所指向的也不是最后一个值而是最后一个值的后一个位置一样。

删除C++中vector的重复元素

res.erase(unique(res.begin(), res.end()), res.end());

排序双指针法

class Solution {
public:
    vector<int> intersection(vector<int>& nums1, vector<int>& nums2) {
        sort(nums1.begin(),nums1.end());
        sort(nums2.begin(),nums2.end());
        vector<int> res;
        for(int i=0,j=0;i<nums1.size() && j<nums2.size();){
            if(nums1[i] == nums2[j]){
                res.push_back(nums1[i]);
                i++;
                j++;
            } else if(nums1[i]<nums2[j]){
                i++;
            }else
                j++;
        }
        res.erase(unique(res.begin(), res.end()), res.end());
        return res;
    }
};

class Solution {
public:
    vector<int> intersection(vector<int>& nums1, vector<int>& nums2) {
        vector<int> res;
        if(nums1.size() ==0 || nums2.size()==0)
            return res;

        sort(nums1.begin(),nums1.end());
        sort(nums2.begin(),nums2.end());

        int lastNum = -1;
        for(int i=0,j=0;i<nums1.size() && j<nums2.size();){
            if(nums1[i] == nums2[j]){
                if(lastNum !=nums1[i]){
                    res.push_back(nums1[i]);
                    lastNum = nums1[i];
                }                
                i++;
            }
            else if(nums1[i]<nums2[j]){
                i++;
            }
            else{
                j++;
            }            
        }
        return res;
    }
};
